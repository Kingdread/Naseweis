def verify(q)
  # Currently only checks if the question type is valid
  if q.is_a? Array
    q.each { |x| verify x }
    return
  end
  type = q['type']
  qs = q['q']
  well = type.nil? || @converter.supported_types.include?(type.intern)
  raise WeisheitError, "invalid type #{type}" unless well
  verify qs if qs.is_a? Array
end
